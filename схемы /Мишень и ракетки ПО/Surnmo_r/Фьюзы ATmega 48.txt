

 Байты конфигурации для  ATmega 48 с кварцем на 4 мгц.

 LOW = DD

CKDIV8		- 1	; Внутренний делитель  тактовой частоты на 8
CKOUT		- 1	; Вывод тактового сигнала на PORTх
SUT1		- 0	; Выбор времени запуска
SUT0		- 1	; Выбор времени запуска
CKSEL3		- 1	; Выбор тактового сигнала
CKSEL2		- 1	; Выбор тактового сигнала
CKSEL1		- 0	; Выбор тактового сигнала
CKSEL0		- 1	; Выбор тактового сигнала

HIGH = DD

RSTDISBL	- 1	; Запрет внешнего сброса ·(PC6·используется для вводя/вывода)
DWEN		- 1	; Включить работу отладочного интерфейса Debug Wire
SPIEN		- 0	; Разрешить внутрисхемное программирование (SPI)
WDTON		- 1	; Сторожевой таймер всегда включен
EESAVE		- 1	; Сохранять содержимое EEPROM при стирании кристалла
BODLEVEL2	- 1	; Порог монитора питания
BODLEVEL1	- 0	; Порог монитора питания
BODLEVEL0	- 1	; Порог монитора питания

EXTENDED = FF

HWBE		- 1	; Hardware Boot Enable
SELFPRGEN	- 1	; Разрешение самопрограммирования
